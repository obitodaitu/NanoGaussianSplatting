NOTE ON WORK ITEM BUFFER TYPE:
  FVisibleClusterWorkItem is 16 bytes = uint4. In HLSL we can declare:
    struct FVisibleClusterWorkItem { uint4 Data; } or use uint4 directly.
  Or use StructuredBuffer with stride=16. Either approach works.
  Recommended: declare as StructuredBuffer<uint4> in C++, unpack in HLSL:
    uint GlobalClusterIndex = WorkList[idx].x;
    uint MetadataIndex      = WorkList[idx].y;
    uint GlobalSplatStart   = WorkList[idx].z;
    uint SplatCount         = WorkList[idx].w;
  This avoids struct alignment issues between C++ and HLSL.

STEP 1.3: Create shader file BuildVisibleClusterWorkList.usf
------------------------------------------------------------------------

Path: Shaders/Private/BuildVisibleClusterWorkList.usf

Contains 3 entry points: ResetCS, LeafCS, LODCS.

  #include "/Engine/Private/Common.ush"
  #include "/Plugin/GaussianSplatting/Private/GaussianDataTypes.ush"

  // Copy FProxyMetadata struct from GlobalCompactSplats.usf (same 128-byte layout)

  StructuredBuffer<FGaussianGPUCluster> GlobalClusterBuffer;
  StructuredBuffer<FProxyMetadata>      ProxyMetadataBuffer;
  StructuredBuffer<uint>               ShadowClusterVisibilityBitmap;
  StructuredBuffer<uint>               ShadowSelectedClusterBuffer;
  StructuredBuffer<uint>               ShadowLODClusterBuffer;
  StructuredBuffer<uint>               ShadowLODClusterCountBuffer;
  RWStructuredBuffer<uint4>            VisibleClusterWorkList;
  RWStructuredBuffer<uint>             VisibleClusterCountBuffer;
  RWStructuredBuffer<uint>             ClusterSplatCountsBuffer;
  uint ProxyCount;
  uint TotalLeafClusterCount;
  uint TotalClusterCount;
  uint UseLODRendering;

  // Binary search: find proxy by global leaf cluster index
  uint FindProxyByLeafCluster(uint globalLeafIdx)
  {
      uint lo = 0, hi = ProxyCount;
      while (lo < hi)
      {
          uint mid = (lo + hi) / 2;
          if (ProxyMetadataBuffer[mid].GlobalLeafClusterEnd <= globalLeafIdx)
              lo = mid + 1;
          else
              hi = mid;
      }
      return lo;
  }

  // Binary search: find proxy by global cluster index (for LOD parents)
  uint FindProxyByCluster(uint globalClusterIdx)
  {
      uint lo = 0, hi = ProxyCount;
      while (lo < hi)
      {
          uint mid = (lo + hi) / 2;
          uint clusterEnd = ProxyMetadataBuffer[mid].GlobalClusterStartIndex
                          + ProxyMetadataBuffer[mid].ClusterCount;
          if (clusterEnd <= globalClusterIdx)
              lo = mid + 1;
          else
              hi = mid;
      }
      return lo;
  }

  [numthreads(1,1,1)]
  void ResetCS(uint3 tid : SV_DispatchThreadID)
  {
      VisibleClusterCountBuffer[0] = 0;
  }

  [numthreads(64,1,1)]
  void LeafCS(uint3 DispatchThreadId : SV_DispatchThreadID)
  {
      uint globalLeafIdx = DispatchThreadId.x;
      if (globalLeafIdx >= TotalLeafClusterCount) return;

      uint proxyIdx = FindProxyByLeafCluster(globalLeafIdx);
      if (proxyIdx >= ProxyCount) return;

      FProxyMetadata metadata = ProxyMetadataBuffer[proxyIdx];

      // Local leaf index within proxy
      uint prevLeafEnd = (proxyIdx > 0)
          ? ProxyMetadataBuffer[proxyIdx-1].GlobalLeafClusterEnd : 0;
      uint localLeafIdx = globalLeafIdx - prevLeafEnd;

      // Global cluster index
      uint globalClusterIdx = metadata.GlobalClusterStartIndex + localLeafIdx;

      // Check visibility bitmap
      uint bitmapWord = metadata.GlobalVisibilityBitmapStart + (localLeafIdx / 32);
      if (!(ShadowClusterVisibilityBitmap[bitmapWord] & (1u << (localLeafIdx % 32))))
          return;

      // Check LOD selection
      if (UseLODRendering)
      {
          uint globalSelectedIdx = metadata.GlobalSelectedClusterStart + localLeafIdx;
          uint packedSelected = ShadowSelectedClusterBuffer[globalSelectedIdx];
          uint selectedClusterID = packedSelected & 0x00FFFFFF;
          if (selectedClusterID != globalClusterIdx)
              return;  // covered by LOD parent
      }

      // Read cluster data
      FGaussianGPUCluster cluster = GlobalClusterBuffer[globalClusterIdx];

      // Append to work list
      uint workIdx;
      InterlockedAdd(VisibleClusterCountBuffer[0], 1, workIdx);

      uint4 item;
      item.x = globalClusterIdx;                                   // GlobalClusterIndex
      item.y = proxyIdx;                                            // MetadataIndex
      item.z = metadata.GlobalSplatStartIndex + cluster.SplatStartIndex; // GlobalSplatStart
      item.w = cluster.SplatCount;                                  // SplatCount
      VisibleClusterWorkList[workIdx] = item;
      ClusterSplatCountsBuffer[workIdx] = cluster.SplatCount;
  }

  [numthreads(64,1,1)]
  void LODCS(uint3 DispatchThreadId : SV_DispatchThreadID)
  {
      uint lodIdx = DispatchThreadId.x;

      // Early-out: LOD cluster count is GPU-driven
      if (lodIdx >= ShadowLODClusterCountBuffer[0]) return;

      uint globalClusterIdx = ShadowLODClusterBuffer[lodIdx];

      // Find owning proxy
      uint proxyIdx = FindProxyByCluster(globalClusterIdx);
      if (proxyIdx >= ProxyCount) return;

      FProxyMetadata metadata = ProxyMetadataBuffer[proxyIdx];
      FGaussianGPUCluster cluster = GlobalClusterBuffer[globalClusterIdx];

      // Append to work list
      uint workIdx;
      InterlockedAdd(VisibleClusterCountBuffer[0], 1, workIdx);

      uint4 item;
      item.x = globalClusterIdx;
      item.y = proxyIdx;
      item.z = metadata.GlobalSplatStartIndex + cluster.LODSplatStartIndex;
      item.w = cluster.LODSplatCount;
      VisibleClusterWorkList[workIdx] = item;
      ClusterSplatCountsBuffer[workIdx] = cluster.LODSplatCount;
  }

DispatchBuildVisibleClusterWorkList():
  EnsureClusterWorkListBuffers();

  // Transition to UAV
  Transition(VisibleClusterWorkList, UAVCompute);
  Transition(VisibleClusterCountBuffer, UAVCompute);
  Transition(ClusterSplatCountsBuffer, UAVCompute);

  // Step 1: Reset
  Set FBuildVisibleClusterWorkListResetCS params.
  Dispatch(1, 1, 1);

  // UAV barrier on VisibleClusterCountBuffer
  Transition(VisibleClusterCountBuffer, UAVCompute -> UAVCompute);

  // Step 2: Leaf clusters
  Set FBuildVisibleClusterWorkListCS params.
  Dispatch(ceil(TotalLeafClusterCount / 64), 1, 1);

  // UAV barrier
  Transition(VisibleClusterCountBuffer, UAVCompute -> UAVCompute);
  Transition(VisibleClusterWorkList, UAVCompute -> UAVCompute);
  Transition(ClusterSplatCountsBuffer, UAVCompute -> UAVCompute);

  // Step 3: LOD parent clusters
  // Need ShadowLODClusterBuffer and ShadowLODClusterCountBuffer as SRV
  // (already transitioned to SRVCompute at end of DispatchGlobalClusterCulling)
  Set FBuildVisibleClusterWorkListLODCS params.
  Dispatch(ceil(TotalClusterCount / 64), 1, 1);
    // Conservative upper bound; shader early-outs if lodIdx >= LODClusterCount

  // Transition outputs to SRV for downstream
  Transition(VisibleClusterWorkList, UAVCompute -> SRVCompute);
  Transition(VisibleClusterCountBuffer, UAVCompute -> SRVCompute);
  Transition(ClusterSplatCountsBuffer, UAVCompute -> SRVCompute);

  StructuredBuffer<uint>   ClusterSplatCountsBuffer;
  StructuredBuffer<uint>   VisibleClusterCountBuffer;
  RWStructuredBuffer<uint> ClusterOutputOffsetsBuffer;
  RWStructuredBuffer<uint> ClusterCalcViewDataIndirectArgs;
  RWStructuredBuffer<uint> TestCalcDistIndirectArgs;
  RWStructuredBuffer<uint> TestSortIndirectArgs;
  RWStructuredBuffer<uint> TestSortParams;
  RWStructuredBuffer<uint> TestDrawIndirectArgs;
  uint MaxRenderBudget;

  [numthreads(1,1,1)]
  void MainCS()
  {
      uint visibleCount = VisibleClusterCountBuffer[0];
      uint offset = 0;

      for (uint i = 0; i < visibleCount; i++)
      {
          ClusterOutputOffsetsBuffer[i] = offset;
          uint splatCount = ClusterSplatCountsBuffer[i];

          if (MaxRenderBudget > 0 && (offset + splatCount) > MaxRenderBudget)
          {
              splatCount = (offset < MaxRenderBudget) ? (MaxRenderBudget - offset) : 0;
          }

          offset += splatCount;

          if (MaxRenderBudget > 0 && offset >= MaxRenderBudget)
          {
              // Zero out remaining
              for (uint j = i + 1; j < visibleCount; j++)
              {
                  ClusterOutputOffsetsBuffer[j] = offset;
              }
              break;
          }
      }

      // Sentinel
      ClusterOutputOffsetsBuffer[visibleCount] = offset;

      // ClusterCalcViewData indirect args (one group per visible cluster)
      ClusterCalcViewDataIndirectArgs[0] = visibleCount;
      ClusterCalcViewDataIndirectArgs[1] = 1;
      ClusterCalcViewDataIndirectArgs[2] = 1;

      // CalcDistances: ceil(TotalVisible / 256)
      TestCalcDistIndirectArgs[0] = (offset + 255) / 256;
      TestCalcDistIndirectArgs[1] = 1;
      TestCalcDistIndirectArgs[2] = 1;

      // RadixSort: ceil(TotalVisible / 1024)
      uint numTiles = (offset + 1023) / 1024;
      TestSortIndirectArgs[0] = numTiles;
      TestSortIndirectArgs[1] = 1;
      TestSortIndirectArgs[2] = 1;

      TestSortParams[0] = offset;
      TestSortParams[1] = numTiles;

      // Draw: {6, instanceCount, 0, 0, 0}
      TestDrawIndirectArgs[0] = 6;
      TestDrawIndirectArgs[1] = offset;
      TestDrawIndirectArgs[2] = 0;
      TestDrawIndirectArgs[3] = 0;
      TestDrawIndirectArgs[4] = 0;
  }

          // Copy transform to groupshared

              gs_LocalToWorld[i*4+1] = meta.LocalToWorld[i].y;
              gs_LocalToWorld[i*4+2] = meta.LocalToWorld[i].z;
              gs_LocalToWorld[i*4+3] = meta.LocalToWorld[i].w;

          gs_OpacityScale = params.OpacityScale;
          gs_SplatScale = params.SplatScale;
          gs_SHOrder = params.SHOrder;
          gs_NumSHCoeffs = params.NumSHCoeffs;
          gs_UseSHRendering = params.UseSHRendering;
          gs_SHBytesPerSplat = params.SHBytesPerSplat;
          gs_GlobalSHByteOffset = meta.GlobalSHByteOffset;
          gs_GlobalSplatStart = globalSplatStart;
          gs_GlobalSplatStartIndex = meta.GlobalSplatStartIndex;
          gs_SplatCount = splatCount;
          gs_OutputOffset = ClusterOutputOffsetsBuffer[GroupID.x];
          gs_GlobalClusterIndex = globalClusterIdx;
          gs_OriginalSplatCount = meta.OriginalSplatCount;
          gs_GlobalSelectedClusterStart = meta.GlobalSelectedClusterStart;
      }
      GroupMemoryBarrierWithGroupSync();

      // --- PER-THREAD SPLAT PROCESSING ---
      for (uint localIdx = GroupThreadID.x;

          localToWorld[2] = float4(gs_LocalToWorld[8], gs_LocalToWorld[9],

          localToWorld[3] = float4(gs_LocalToWorld[12], gs_LocalToWorld[13],

          float opacityScale = gs_OpacityScale;

          FUnpackedSplat splat = UnpackSplat(GlobalPackedSplatBuffer, globalSplatIdx);

          if (clipPos.w <= 0.0)
